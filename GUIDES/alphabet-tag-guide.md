<!-- forces the text to become bold -->
<bold> bold text </bold>

<!-- forces the text to become plain (not-bold) -->
<plain> plain text </plain>

<!-- paints the text -->
<color value=#RRGGBB> colored text </color>

<!-- makes the text all colorful! (uniform: syncs the characters to have same color) -->
<rainbow speed=3 uniform=false> rainbow text </rainbow>

<!-- shakes the text -->
<shake speed=1 intensity=5> shakey text </shake>

<!-- makes the text wavey -->
<wave speed=1 intensity=5> wavey text </wave>
