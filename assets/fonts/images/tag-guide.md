<!-- forces the text inside the tag to become bold -->
<bold> bold text </bold>

<!-- forces the text inside the tag to become plain (not-bold) -->
<plain> plain text </plain>

<!-- paints the text inside the tag -->
<color value=#RRGGBB> colored text </color>

<!-- shakes the text inside the tag -->
<shake speed=1 intensity=5> shakey text </shake>

<!-- makes the text inside the tag wavey -->
<wave speed=1 intensity=5> wavey text </wave>
